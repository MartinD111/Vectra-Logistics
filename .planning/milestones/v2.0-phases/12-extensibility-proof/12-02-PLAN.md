import type { PluginBlockManifest } from './manifest';

const dedupe: PluginBlockManifest = { id: 'vectra.dedupe', /* ... */ };

const wordCount: PluginBlockManifest = {
  id: 'vectra.wordcount',
  version: '1.0.0',
  title: 'Word count column',
  description: 'Add a column with the number of words in a chosen text column.',
  icon: 'Hash',
  group: 'process',
  origin: 'builtin',
  settingsSchema: [
    { key: 'column', label: 'Text column', type: 'column' },
    { key: 'into', label: 'New column name', type: 'text', default: 'word_count' },
  ],
  uiSchema: [
    { node: 'text', text: 'Counts words in "{{config.column}}" → "{{config.into}}".', variant: 'muted' },
  ],
  logic: {
    kind: 'transform',
    source: `
      var col = config.column;
      var into = config.into || 'word_count';
      if (!col) return { rows: rows };
      return rows.map(function (r) {
        var copy = {};
        for (var k in r) copy[k] = r[k];
        var text = r[col] == null ? '' : String(r[col]).trim();
        copy[into] = text ? text.split(/\\s+/).length : 0;
        return copy;
      });
    `,
  },
};

export const EXAMPLE_PLUGINS: PluginBlockManifest[] = [dedupe, wordCount];
```

interface Ctx { config: Record<string, unknown>; vars: Record<string, unknown>; count: number; }

function resolve(tpl: string, ctx: Ctx): string {
  return tpl.replace(/\{\{\s*([\w.]+)\s*\}\}/g, (_, path: string) => {
    if (path === 'count') return String(ctx.count);
    if (path.startsWith('config.')) return str(ctx.config[path.slice(7)]);
    if (path.startsWith('vars.')) return str(ctx.vars[path.slice(5)]);
    return '';
  });
}
// ctx built as: { config: block.config, vars: rt.state.vars, count: dataset.length }
// where dataset = rt.result.outputOf[block.id] ?? rt.result.inputTo[block.id] ?? []
```
This confirms `{{count}}` in a `text`/`badge` node's `text` template resolves to the current dataset's row count automatically — no runtime/registry change needed for this plugin.

The sandboxed `logic.source` runs in a hardened Web Worker (`runInSandbox`, per 12-RESEARCH.md Security Domain) with `rows`, `config`, `vars` in scope for `transform` logic — no DOM/network access.
