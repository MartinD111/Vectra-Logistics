let cachedDeploymentMode: DeploymentMode | undefined;

export function validateDeploymentModeValue(value: string | undefined): SecretValidationResult { ... }

export function getDeploymentMode(): DeploymentMode {
  if (cachedDeploymentMode !== undefined) return cachedDeploymentMode;
  const value = process.env.DEPLOYMENT_MODE;
  const result = validateDeploymentModeValue(value);
  if (!result.valid) fail('DEPLOYMENT_MODE', result.reason ?? 'invalid value');
  cachedDeploymentMode = value as DeploymentMode;
  return cachedDeploymentMode;
}

    - `resolveVersion(envValue, readVersionFile)`: given `envValue = '3.0.0'` (or any non-empty trimmed string) returns `'3.0.0'` without calling `readVersionFile`.
    - `resolveVersion(undefined, readVersionFile)` where `readVersionFile` returns `'3.0.0\n'` returns `'3.0.0'` (trimmed).
    - `resolveVersion(undefined, readVersionFile)` where `readVersionFile` throws returns `'unknown'`.
    - `resolveVersion('', readVersionFile)` (empty string, falsy) falls through to `readVersionFile` exactly like `undefined`.
    - `getVersion()` caches: set `process.env.VERSION = 'cached-val'`, call `getVersion()` once, mutate `process.env.VERSION = 'other-val'`, call `getVersion()` again — both calls return `'cached-val'` (mirrors `secrets.test.ts`'s `getDeploymentMode()` caching test).
  </behavior>
  <action>

    Create `apps/api/src/core/config/version.ts` mirroring `secrets.ts`'s "pure function + cached wrapper" shape:
    - Export `resolveVersion(envValue: string | undefined, readVersionFile: () => string): string` — pure, no module-level state, no `fs` import needed in its signature (takes `readVersionFile` as a parameter so it's directly unit-testable without touching the real filesystem). Logic: if `envValue` is truthy after `.trim()`, return the trimmed value; otherwise call `readVersionFile()`, trim its result, and return it; if `readVersionFile()` throws, catch and return `'unknown'`.

Never calls `process.exit` — this is a soft, non-fatal read (unlike `secrets.ts`'s `fail()`), matching RESEARCH.md's `'unknown'` fallback design.
