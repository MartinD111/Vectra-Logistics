```typescript
export interface SecretValidationResult {
  valid: boolean;
  reason?: string;
}

export function validateJwtSecretValue(value: string | undefined): SecretValidationResult { ... }
export function validateEncryptionKeyValue(value: string | undefined): SecretValidationResult { ... }

function fail(varName: string, reason: string): never {
  console.error(`FATAL: invalid ${varName} — ${reason}. Set a real, non-default value before starting the server.`);
  process.exit(1);
}

export function getJwtSecret(): string { ... }
export function validateSecretsOrExit(): void {
  getJwtSecret();
  const encValue = process.env.ENCRYPTION_KEY;
  const encResult = validateEncryptionKeyValue(encValue);
  if (!encResult.valid) { fail('ENCRYPTION_KEY', encResult.reason ?? 'invalid value'); }
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|--------------|
| Public internet -> POST /api/auth/signup | Untrusted, unauthenticated request body reaches an endpoint that creates companies + admin users |
| Process env -> secrets.ts | `DEPLOYMENT_MODE` is a trusted (operator-set) input, not user input, but must still be validated against exact expected values |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|------------------|
| T-16-01 | Elevation of Privilege | `POST /api/auth/signup` | mitigate | D-04: unconditional 403 when `DEPLOYMENT_MODE='on-prem'`, checked before any DB write — prevents an attacker on a customer's on-prem box from creating additional companies/admin accounts via the public signup endpoint (V4 Access Control) |
| T-16-02 | Tampering | `DEPLOYMENT_MODE` boot-time read | mitigate | Exact-match validation (`'cloud'`/`'on-prem'` only) + fail-fast `process.exit(1)` on anything else — mirrors the existing SEC-01/SEC-02 pattern so a misconfigured or garbage value can never silently fall through to either behavior (V5 Input Validation on a trusted-but-unvalidated config surface) |
| T-16-03 | Information Disclosure (via inconsistent behavior) | `getDeploymentMode()` per-request re-evaluation | mitigate | Caching after first read (Task 1) — prevents a scenario where an env-var change mid-process (e.g. via a compromised orchestration layer) could flip access-control behavior for a subset of in-flight requests; DEP-02 success criterion 4 requires read-once |

</threat_model>
